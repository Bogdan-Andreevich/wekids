<script>$(document).ready(function() {
    if($(".headertypeahead").length) {
        $.typeahead({
            input: ".headertypeahead",
            order: "asc",
            dynamic: true,
            delay: 500,
            minLength: 2,
            maxItem: 5,
            hint: false,
            filter: false,
            //emptyTemplate: translate('NoResultsFoundFor', catalogVocabulary) + ": {{query}}",
            source: {
                ajax: function (query) {
                    return {
                        type: "GET",
                        url: "/catalog/catalog/list",
                        data: {
                            input: "{{query}}"
                        },
                        callback: {
                            done: function (data) {
                                return data;
                            }
                        }
                    };
                }
            },
            callback: {
                onClick: function (node, a, item, event) {
                    window.location = '/catalog?name='+item.display;
                }
            },
        });
        
        document.querySelector('.headertypeahead').addEventListener('keyup', function (event) {
            if (event.keyCode == 13) {
                window.location = '/catalog?name='+document.querySelector('.headertypeahead').value;
            }
        });
        
        document.querySelector('.typeahead__submit').addEventListener('click', function (event) {
            window.location = '/catalog?name='+document.querySelector('.headertypeahead').value;
            $('.headertypeahead').blur();
        });
    }
});</script></body>
